17   public class groupPhotoListAdapter extends ArrayAdapter<String>{ 
18       private final Activity context; 
19       private final String[] titles; 
20       private final String[] comments; 
21       private final Bitmap[] bitmap; 
22       private int width; 
23       private int height; 
24       public groupPhotoListAdapter(Activity context, 
25                                    String[] titles, String[] comments, Bitmap[] bitmap) { 
26           super(context, R.layout.photo_row, titles); 
27           this.context = context; 
28           this.titles = titles; 
29           this.comments = comments; 
30           this.bitmap = bitmap; 
31           this.width = 10; 
32           this.height = 10; 
33       } 
34       @Override 
35       public View getView(int position, View view, ViewGroup parent) { 
36           LayoutInflater inflater = context.getLayoutInflater(); 
37           View rowView= inflater.inflate(R.layout.photo_row, null, true); 
38           TextView txtTitle = (TextView) rowView.findViewById(R.id.photoListTitle); 
39           TextView txtComment= (TextView) rowView.findViewById(R.id.photoListSubTitle); 
40           ImageView imageView = (ImageView) rowView.findViewById(R.id.photoListPhoto); 
41           txtTitle.setText(titles[position]); 
42           txtComment.setText(comments[position]); 
43           Bitmap resized = Bitmap.createScaledBitmap(bitmap[position], width, height, true); 
44           imageView.setImageBitmap(resized); 
45           return rowView; 
46       } 
47    
48    
49       public void setSizes(int w, int h) 
50       { 
51           this.width = w; 
52           this.height = h; 
53       }
